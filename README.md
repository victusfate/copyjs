copyjs is for deep copies of javascript objects
===

<pre>
npm install copyjs
</pre>

	var a = {
	  val: 15,
	  business: 'isGood'
	};

	var b = copy(a);

    #  object a { val: 15, business: 'isGood' }
    #  object b { val: 15, business: 'isGood' }

	b.val = 8000;
	b.business = 'isBetter';

	#  object a { val: 15, business: 'isGood' }
	#  object b { val: 8000, business: 'isBetter' }
